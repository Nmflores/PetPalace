-- SELECT ALL WORKERS -- ROLE ID 1
SELECT USERS.FIRST_NAME, ROLES.ROLE FROM USERS, ROLES WHERE USERS.USER_ROLE = 1 AND ROLES.ROLE_ID = 1;

-- SELECT ALL CLIENTS -- ROLE ID 2
SELECT USERS.FIRST_NAME, ROLES.ROLE FROM USERS, ROLES WHERE USERS.USER_ROLE = 2 AND ROLES.ROLE_ID = 2;


-- SELECT ALL USERS THAT HAVE A WORK
SELECT A.first_name, B.role, C.service_name FROM USERS A, ROLES B, AVAILABLE_SERVICES C, WORKER_SERVICES D WHERE A.user_role = B.role_id AND D.worker_id = A.user_id AND D.service_id = C.service_id;



-- SELECT FEMALE WORKERS -- user_gender = 1 AND user_role = 1
